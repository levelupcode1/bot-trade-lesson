# 업비트 API 키 설정
# 실제 사용 시 이 파일을 .env로 복사하고 실제 키를 입력하세요

# 업비트 API Access Key
UPBIT_ACCESS_KEY=your_access_key_here

# 업비트 API Secret Key  
UPBIT_SECRET_KEY=your_secret_key_here

# API 기본 URL (변경하지 마세요)
UPBIT_BASE_URL=https://api.upbit.com

# 요청 타임아웃 (초)
UPBIT_TIMEOUT=30

# 최대 재시도 횟수
UPBIT_MAX_RETRIES=3
